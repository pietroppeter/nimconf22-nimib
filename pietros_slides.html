nb.partials["nbCodeHtmlOutput"] =
  nb.partials["nbCode"].replace("{{>nbCodeOutput}}", "{{&output}}")

Composing other blocks

template nbTextRepeat(text: string, repeat: int) =
  for i in 1 .. repeat:
    nbText: text

nbTextRepeat("All work and no play makes Jack a dull boy", 3)
